229void Main_voidTimer0ISR(void)
230{
231 static uint16 Local_u16Counter;
232
233 Local_u16Counter++;
234
235 /*Control the LDRs sampling flag*/
236 if((Local_u16Counter % 80) == 0u)
237 {
238 Main_u8LDRSamplingFlag = MAIN_u8SAMPLING_READY;
239 }
240
241 /*Control the rear ultrasonic distance sampling flag*/
242 if((Local_u16Counter % 1000) == 0u)
243 {
244 Main_u8UltrasonicSamplingFlag = MAIN_u8SAMPLING_READY;
245 }
246
247 /*Control the temperature sampling flag*/
248 if(Local_u16Counter == 2000u)
249 {
250 Main_u8FuelLevelSamplingFlag = MAIN_u8SAMPLING_READY;
251 Main_u8TempSamplingFlag = MAIN_u8SAMPLING_READY;
252 Local_u16Counter = 0;
253 }
