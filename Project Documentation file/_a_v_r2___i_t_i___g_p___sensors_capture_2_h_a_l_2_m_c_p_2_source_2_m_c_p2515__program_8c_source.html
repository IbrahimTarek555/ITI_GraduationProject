127void MCP2515_voidSendCANmsg(MCP2515_CanMessage_t Copy_stMessage)
128{
129 MCP2515_voidChipSelect(SLAVE_CHOOSE);
130
131 uint8 Local_u8Counter;
132
133 /*Send header and address*/
134 SPI_u8Transceive(INST_WRITE);
135 SPI_u8Transceive(TXB0CTRL);
136
137 /*Setup message priority*/
138 SPI_u8Transceive(3);
139
140 /* Setup standard or extended identifier */
141 SPI_u8Transceive((uint8)(Copy_stMessage.id >> 3));
142 SPI_u8Transceive((uint8)(Copy_stMessage.id << 5));
143 SPI_u8Transceive(0);
144 SPI_u8Transceive(0);
145
146 /* Setup message length and RTR bit */
147 SPI_u8Transceive((Copy_stMessage.length) | (Copy_stMessage.properties << 6));
148
149 /* Store the message into the buffer */
150 for(Local_u8Counter = 0; Local_u8Counter < Copy_stMessage.length; Local_u8Counter++)
151 {
152 SPI_u8Transceive(Copy_stMessage.data[Local_u8Counter]);
153 }
154
155 /*Release the bus*/
156 MCP2515_voidChipSelect(SLAVE_RELEASE);
157
158 /*Send request to send*/
159 MCP2515_voidChipSelect(SLAVE_CHOOSE);
160 SPI_u8Transceive(INST_RTS_TXB0);
161 MCP2515_voidChipSelect(SLAVE_RELEASE);
162}

225MCP2515_CanMessage_t MCP2515_stReceiveCANmsg(void)
226{
227 MCP2515_CanMessage_t Local_stReceive = {0};
228 uint8 Local_u8Counter = 0;

242
243 MCP2515_voidWriteRegister(CANINTF, 0);
244

247 uint8 Local_u8Hegazy = MCP2515_u8ReadBuffer[2];
248 Local_stReceive.id |= Local_u8Hegazy >> 5;
249 Local_stReceive.length = MCP2515_u8ReadBuffer[5] & 0x0F;
250
251 for(Local_u8Counter = 0; Local_u8Counter < 8; Local_u8Counter++)
252 {
253 Local_stReceive.data[Local_u8Counter] = MCP2515_u8ReadBuffer[6 + Local_u8Counter];
254 }
255
256 Local_stReceive.properties = GET_BIT(MCP2515_u8ReadBuffer[0],3);
