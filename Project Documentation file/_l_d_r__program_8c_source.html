35uint8 LDR_u8GetAnalogVolt(uint8 Copy_u8ADCChannel, uint16 *Copy_pu16AnalogVolt)
36{
37 /*Define variables*/
38 uint8 Local_u8ErrorState = OK;
39 uint16 Local_u16ADCReading = 0;
40
41 /*Validate the pointer passed by the user*/
42 if(Copy_pu16AnalogVolt != NULL)
43 {
44 /*Get the reading from ADC*/
45 switch(Copy_u8ADCChannel)
46 {
47 case LDR_u8ADC_CH0: ADC_u8StartConversionSynch(ADC_u8SINGLE_ENDED_CH0, &Local_u16ADCReading); break;
48 case LDR_u8ADC_CH1: ADC_u8StartConversionSynch(ADC_u8SINGLE_ENDED_CH1, &Local_u16ADCReading); break;
49 case LDR_u8ADC_CH2: ADC_u8StartConversionSynch(ADC_u8SINGLE_ENDED_CH2, &Local_u16ADCReading); break;
50 case LDR_u8ADC_CH3: ADC_u8StartConversionSynch(ADC_u8SINGLE_ENDED_CH3, &Local_u16ADCReading); break;
51 case LDR_u8ADC_CH4: ADC_u8StartConversionSynch(ADC_u8SINGLE_ENDED_CH4, &Local_u16ADCReading); break;
52 case LDR_u8ADC_CH5: ADC_u8StartConversionSynch(ADC_u8SINGLE_ENDED_CH5, &Local_u16ADCReading); break;
53 case LDR_u8ADC_CH6: ADC_u8StartConversionSynch(ADC_u8SINGLE_ENDED_CH6, &Local_u16ADCReading); break;
54 case LDR_u8ADC_CH7: ADC_u8StartConversionSynch(ADC_u8SINGLE_ENDED_CH7, &Local_u16ADCReading); break;
55 default: Local_u8ErrorState = NOK;
56 }
57
58 /*Calculate the voltage and return it to the user*/
59#if LDR_u8ADC_RESOLUTION == LDR_u8ADC_8BITS_RES
60 *Copy_pu16AnalogVolt = (uint16) (((uint32) Local_u16ADCReading * LDR_u8ADC_REF_VOLT * 1000UL) / 256UL);
61#elif LDR_u8ADC_RESOLUTION == LDR_u8ADC_10BITS_RES
62 *Copy_pu16AnalogVolt = (uint16) (((uint32) Local_u16ADCReading * LDR_u8ADC_REF_VOLT * 1000UL) / 1024UL);
63#else
64#error Wrong LDR_u8ADC_RESOLUTION configuration option
65#endif
66
67 }
68 else
69 {
70 Local_u8ErrorState = NULL_PTR_ERR;
71 }
72
73 /*Return the error state*/
74 return Local_u8ErrorState;
75}
