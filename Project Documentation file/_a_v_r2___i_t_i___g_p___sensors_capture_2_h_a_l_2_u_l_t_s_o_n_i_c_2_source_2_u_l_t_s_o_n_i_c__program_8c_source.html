42uint8 ULTSONIC_u8MeasureDistance(const ULTSONIC_t *Copy_pstUltsonicData, uint16 *Copy_pu16Distance)
43{
44 /*Define Variables*/
45 uint8 Local_u8ErrorState = OK;
46 uint8 Local_u8ErrorFounder = OK;
47 uint32 Local_u32EchoTime = 0u;
48 uint32 Local_u32TimeoutCounter = 0u;
49
50 /*Initialize timer 1*/
51 TIMER1_CFG_t Local_stTimer1Cfg;
52 Local_stTimer1Cfg.ClkSelect = TIMER1_u8SYS_CLK_BY_8;
53 Local_stTimer1Cfg.WGM = NORMAL_16_BITS;
54 Local_stTimer1Cfg.OCAMode = OC_DISCONNECTED;
55 Local_stTimer1Cfg.OCBMode = OC_DISCONNECTED;
56 Local_u8ErrorFounder = TIMER1_u8Init(&Local_stTimer1Cfg);
57
58 if(Local_u8ErrorFounder == OK)
59 {
60 /*Initialize and Enable Capture Unit and Overflow Interrupts of timer 1*/
61 TIMERS_u8SetCallBack(TIMER1_OVF, ULTSONIC_voidOvfISR);
62 TIMER1_u8SetInputCaptTriggerSrc(TIMER1_u8ICU_RISING_EDGE);
63 TIMERS_u8SetCallBack(TIMER1_CAPT, ULTSONIC_voidCaptureISR);
64 TIMERS_u8IntEnable(TIMER1_CAPT);
65
66 /*Trigger the ultrasonic to return its echo*/
67 DIO_u8SetPinValue(Copy_pstUltsonicData -> TrigPort, Copy_pstUltsonicData -> TrigPin, DIO_u8PIN_HIGH);
68 _delay_us(100);
69 DIO_u8SetPinValue(Copy_pstUltsonicData -> TrigPort, Copy_pstUltsonicData -> TrigPin, DIO_u8PIN_LOW);
70
71 /*Polling until the echo time is measured*/
72 while(((ULTSONIC_u16Capture1 == 0u) || (ULTSONIC_u16Capture2 == 0u)) && (Local_u32TimeoutCounter < ULTSOINC_u32TIMEOUT))
73 {
74 Local_u32TimeoutCounter++;
75 }
76
77 /*Disable timer 1 and overflow interrupt of it*/
78 TIMERS_u8IntDisable(TIMER1_OVF);
79 TIMERS_u8IntDisable(TIMER1_CAPT);
80 TIMERS_u8Disable(TIMER1);
81
82 /*Check if there was overflow interrupts or not*/
83 if(ULTSONIC_u16EchoOvfCounter == 0u)
84 {
85 /*Calculate Echo Time in microseconds*/
86 Local_u32EchoTime = ULTSONIC_u16Capture2 - ULTSONIC_u16Capture1;
87 }
88 else
89 {
90 /*Calculate Echo Time in microseconds*/
91 if(ULTSONIC_u16Capture2 >= ULTSONIC_u16Capture1)
92 {
93 Local_u32EchoTime = (ULTSONIC_u16Capture2 - ULTSONIC_u16Capture1) + (ULTSONIC_u16EchoOvfCounter * 65536u);
94 }
95 else if(ULTSONIC_u16Capture2 < ULTSONIC_u16Capture1)
96 {
97 Local_u32EchoTime = (ULTSONIC_u16Capture1 - ULTSONIC_u16Capture2) + (ULTSONIC_u16EchoOvfCounter * 65536u);
98 }
99 }
100
101 /*Measure Distance using Echo Time*/
102 *Copy_pu16Distance = (((f32) Local_u32EchoTime / 2.0) * (34.0 / 1000.0));
103
104 /*Reset the global variables*/
105 ULTSONIC_u16Capture1 = 0u;
106 ULTSONIC_u16Capture2 = 0u;
107 ULTSONIC_u16OvfCounter = 0u;
108 ULTSONIC_u16EchoOvfCounter = 0u;
109 }
110 else
111 {
112 Local_u8ErrorState = NOK;
113 }
114
115 /*Return Error State*/
116 return Local_u8ErrorState;
117}

122static void ULTSONIC_voidCaptureISR(void)
123{
124 static uint8 Local_u8Counter = 0u;
125
126 Local_u8Counter++;
127
128 if(Local_u8Counter == 1u)
129 {
130 /*Enable Overflow interrupt of timer1*/
131 TIMERS_u8IntEnable(TIMER1_OVF);
132 ULTSONIC_u16Capture1 = TIMER1_u16GetInputCaptValue();
133 TIMER1_u8SetInputCaptTriggerSrc(TIMER1_u8ICU_FALLING_EDGE);
134 }
135 else if(Local_u8Counter == 2u)
136 {
137 ULTSONIC_u16Capture2 = TIMER1_u16GetInputCaptValue();
138 ULTSONIC_u16EchoOvfCounter = ULTSONIC_u16OvfCounter;
139 Local_u8Counter = 0;
140 }
141}
