5def create_gauge(canvas, x, y, radius, color, label, start_value, end_value):
6 """
7 Create a gauge on the canvas.
8
9 Args:
10 canvas (Tkinter.Canvas): The canvas to draw on.
11 x (int): The x-coordinate of the center of the gauge.
12 y (int): The y-coordinate of the center of the gauge.
13 radius (int): The radius of the gauge.
14 color (str): The color of the gauge outline.
15 label (str): The label of the gauge.
16 start_value (float): The start value of the gauge arc.
17 end_value (float): The end value of the gauge arc.
18 """
19 # Draw gauge outline
20 canvas.create_arc(x - radius, y - radius, x + radius, y + radius, start=start_value, extent=end_value - start_value,
21 style=tk.ARC, outline=color, width=10)
22 # Draw gauge label
23 canvas.create_text(x, y + radius + 25, text=label, font=("Arial", 12, "bold"), fill="red")
